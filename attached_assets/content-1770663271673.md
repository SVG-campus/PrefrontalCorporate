        n = 5000
        df = pd.DataFrame({
            'Temp_C': np.random.uniform(150, 300, n),

        best = df.loc[df['Strength'].idxmax()]

    def optimize_aerodynamics(self):
        print(">> [PHASE 1] Simulating Aerodynamic Geometries (CFD Approx)...")
        # Shape Factor: 1.0=Sphere, 6.0=Teardrop
        n = 5000
        df = pd.DataFrame({'Fineness_Ratio': np.random.uniform(1, 10, n)})

        # Drag Coef (Cd) Model: Sphere=0.47, Teardrop=0.04
        df['Cd'] = 0.47 * np.exp(-0.5 * (df['Fineness_Ratio'] - 3)**2) + 0.04

        A = 0.05 / df['Fineness_Ratio']

        # Mycelium disintegrates > Mach 0.85 (~290 m/s)
        df['Velocity'] = df['Velocity'].clip(upper=285.0)

        best = df.loc[df['Velocity'].idxmax()]
        return best
