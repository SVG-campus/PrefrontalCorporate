def get_augmented_data():
    print("ðŸ“¦ [NEXUS] Downloading & Augmenting Data...")
    url = "https://archive.ics.uci.edu/ml/machine-learning-databases/00320/student.zip"
    try:
        r = requests.get(url)
        z = zipfile.ZipFile(io.BytesIO(r.content))
        df_mat = pd.read_csv(z.open('student-mat.csv'), sep=';')
        df_por = pd.read_csv(z.open('student-por.csv'), sep=';')
        df = pd.concat([df_mat, df_por], ignore_index=True)

        # Preprocessing
        mapping = {'yes': 1, 'no': 0, 'GP': 1, 'MS': 0, 'F': 1, 'M': 0, 'U': 1, 'R': 0, 'LE3': 0, 'GT3': 1, 'T': 1, 'A': 0}
        for col in df.select_dtypes(include='object').columns:
            if df[col].nunique() == 2:
                df[col] = df[col].map(mapping).fillna(0)
            else:
                df = pd.get_dummies(df, columns=[col], drop_first=True)

        # Convert to US Scale
        def to_us_scale(g):
            if g < 10: return g * 6
            return 60 + (g - 10) * 4
        df['G3'] = df['G3'].apply(to_us_scale)

        # *** SYNTHETIC INJECTION ***
        # We generate "Perfect Students" to teach the model what perfection looks like.
        print(" ðŸ§ª Injecting 500 'Titan' Students into Training Set...")

        n_titans = 500
        titan_data = pd.DataFrame(columns=df.columns)

        # Fill Titans with Perfect Habits
        # We use the mean of the real data for context variables (school, age, etc.)
        # But we MAXIMIZE the habits.
        base_titan = df.mean().to_dict()

        for i in range(n_titans):
            t = base_titan.copy()
            # TITAN HABITS:
            t['studytime'] = 4.0 + np.random.normal(0, 0.1) # Max Study
            t['health'] = 5.0 # Max Health
            t['failures'] = 0 # Zero Failures
            t['absences'] = 0 # Zero Absences
            t['Dalc'] = 1.0 # Zero Alcohol
            t['Walc'] = 1.0
            t['goout'] = 2.0 # Moderate Social
            t['freetime'] = 2.0 # Moderate Free time
            t['G3'] = 100.0 # PERFECT GRADE

            # Add some noise to context variables to prevent overfitting
            t['age'] += np.random.choice([-1, 0, 1])

            titan_data = pd.concat([titan_data, pd.DataFrame([t])], ignore_index=True)

        # Merge Real + Titans
        df_augmented = pd.concat([df, titan_data], ignore_index=True)
        print(f" âœ“ Data Ready: {len(df)} Real + {n_titans} Titans = {len(df_augmented)} Total")

        return df_augmented
    except Exception as e:
        print(e)
        return None

class TitanEngine:
    def __init__(self, df):
        self.df = df
        self.model = None
        self.cols = []

    def train(self):
        print("ðŸ§  [CORTEX] Training Hybrid Causal Stack...")

        # Using a powerful ensemble that can learn the "Titan" pattern
        estimators = [\
            ('xgb', xgb.XGBRegressor(n_estimators=1000, max_depth=6, learning_rate=0.05, n_jobs=1, random_state=42)),\
            ('lgb', lgb.LGBMRegressor(n_estimators=1000, num_leaves=31, learning_rate=0.05, n_jobs=1, random_state=42, verbose=-1)),\
            ('rf', RandomForestRegressor(n_estimators=200, n_jobs=1, random_state=42))\
        ]

        stack = StackingRegressor(estimators=estimators, final_estimator=LinearRegression(), cv=5)

        self.model = make_pipeline(
            RobustScaler(),
            stack
        )

        X = self.df.drop(['G1', 'G2', 'G3'], axis=1)
        y = self.df['G3']
        self.cols = X.columns.tolist()

        self.model.fit(X, y)
        print(f"ðŸ† MODEL ACCURACY (RÂ²): {self.model.score(X, y):.5f}")

    def predict(self, vector_df):
        missing = set(self.cols) - set(vector_df.columns)
        for c in missing: vector_df[c] = 0
        return self.model.predict(vector_df[self.cols])[0]

def optimize_habits(engine, df_template):
    print("\nðŸ§¬ [EVOLUTION] Breeding Perfect Daily Schedule...")
    base_row = df_template.drop(['G1', 'G2', 'G3'], axis=1).mean().to_frame().T

    # We now search for the "Titan" configuration
    bounds = [(3.5, 4.5), (4.5, 5.0), (1.5, 2.5), (1.5, 2.5), (0.0, 0.5), (1.0, 1.1)]

    def objective(x):
        try:
            trial = base_row.copy()
            trial['studytime'] = x[0]
            trial['health'] = x[1]
            trial['goout'] = x[2]
            trial['freetime'] = x[3]
            trial['absences'] = x[4]
            trial['Dalc'] = x[5]

            grade = engine.predict(trial)
            return -grade
        except:
            return 100.0

    result = differential_evolution(objective, bounds, strategy='best1bin', maxiter=50, popsize=20, mutation=(0.5, 1))
    return result.x, -result.fun
